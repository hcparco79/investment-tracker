function doGet(e) { try { var p = e && e.parameter ? e.parameter : {}; // 연결 테스트 if (p.test && !p.name) { return response({result: 'ok', message: '연결 성공!'}); } var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('투자기록'); if (!sheet) { ss.getActiveSheet().setName('투자기록'); sheet = ss.getActiveSheet(); } // 헤더 자동 생성 if (sheet.getLastRow() === 0) { sheet.appendRow([ '날짜','종류','국내/해외','증권사','종목명', '금액(원)','금액(USD)','수량','단가','계좌번호','기록일시' ]); } // 숫자 변환 헬퍼 (콤마 제거 후 숫자로) function toNum(v) { if (!v) return ''; var n = parseFloat(String(v).replace(/,/g, '')); return isNaN(n) ? '' : n; } var row = [ p.date || '', p.type || '', p.market || '', p.broker || '', p.name || '', toNum(p.amountKRW), toNum(p.amountUSD), toNum(p.quantity), toNum(p.price), p.account || '', Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy.MM.dd HH:mm:ss') ]; sheet.appendRow(row); // 금액(원) F열, 금액(USD) G열 서식 적용 var lastRow = sheet.getLastRow(); if (p.amountKRW) { sheet.getRange(lastRow, 6).setNumberFormat('#,##0'); } if (p.amountUSD) { sheet.getRange(lastRow, 7).setNumberFormat('#,##0.0000'); } if (p.quantity) { sheet.getRange(lastRow, 8).setNumberFormat('#,##0'); } if (p.price) { sheet.getRange(lastRow, 9).setNumberFormat('#,##0'); } SpreadsheetApp.flush(); // 즉시 저장 강제 return response({result: 'success', rows: sheet.getLastRow()}); } catch(err) { return response({result: 'error', message: err.toString()}); } } function response(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
